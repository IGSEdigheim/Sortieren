bubble_sort (Liste A)
    n = L채nge von A
    solange n > 1
    wiederhole
        i = 0
        solange i < n-1
        wiederhole
            wenn A[i] > A[i+1] dann
                tausche A[i] mit A[i+1]
            ende wenn
            i = i+1
        ende wiederhole
        n = n-1
    ende wiederhole


selection_sort (Liste A)
    n = L채nge von A
    i = 0
    solange i < n-1
    wiederhole
        kleinste = i
        j = i+1
        solange j < n
        wiederhole
            wenn A[j] < A[kleinste] dann
                kleinste = j
            ende wenn
            j = j+1
        ende wiederhole
        tausche A[kleinste] mit A[i]
        i = i+1
    ende wiederhole

insertion_sort (Liste A)
    n = L채nge von A
    i = 1
    solange i < n
    wiederhole
        merke = A[i]
        j = i
        solange j > 0 und A[j-1] > merke
        wiederhole
            A[j] = A[j-1]
            j = j-1
        ende wiederhole
        A[j] = merke
        i = i+1
    ende wiederhole
